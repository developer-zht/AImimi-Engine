### 1. 分层处理

```typescript
// ✅ 推荐：在不同层次捕获不同错误
async function loadEnvironment(path: string) {
  try {
    const texture = await createHDRTexture(gl, path)
    return texture
  } catch (error) {
    if (error instanceof TextureLoadError) {
      // 可以重试
      console.warn('加载失败，使用备用纹理')
      return await createHDRTexture(gl, 'default.hdr')
    } else if (error instanceof InvalidTextureFormatError) {
      // 数据格式错误，无法恢复
      console.error('数据格式错误:', error.toUserMessage())
      throw error
    } else if (error instanceof TextureCreationError) {
      // GPU 资源问题，尝试降低质量
      console.warn('创建失败，尝试降低分辨率')
      return await createLowResTexture(gl, path)
    } else {
      // 未知错误
      console.error('未知错误:', error)
      throw error
    }
  }
}
```

### 2. 用户友好的错误提示

```typescript
// ✅ 根据错误类型显示不同提示
function showErrorDialog(error: Error) {
  if (error instanceof TextureLoadError) {
    showDialog({
      title: '加载失败',
      message: error.toUserMessage(),
      actions: [
        { label: '重试', onClick: () => retry() },
        { label: '使用默认', onClick: () => useDefault() }
      ]
    })
  } else if (error instanceof TextureCreationError) {
    showDialog({
      title: 'GPU 内存不足',
      message: error.toUserMessage(),
      actions: [
        { label: '降低质量', onClick: () => lowerQuality() },
        { label: '关闭', onClick: () => close() }
      ]
    })
  } else {
    showDialog({
      title: '错误',
      message: '发生未知错误，请刷新页面重试。',
      actions: [{ label: '刷新', onClick: () => location.reload() }]
    })
  }
}
```

### 3. 日志和监控

```typescript
// ✅ 统一上报错误
function reportError(error: Error) {
  if (error instanceof TextureError) {
    const errorData = error.toJSON()

    // 发送到日志系统
    logger.error('TextureError', errorData)

    // 发送到监控系统
    monitor.track('texture_error', {
      code: error.code,
      type: error.name,
      recoverable: error.recoverable,
      context: error.context
    })
  }
}
```
